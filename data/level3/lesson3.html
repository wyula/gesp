01 字符数组

字符串是由0个或多个字符组成的有序序列。在C++中，字符串可以用字符数组来表示。


字符串的特点：

1. 字符串必须以'\0'结尾

2. '\0'是字符串结束的标志，其ASCII码值为0

3. 字符数组的长度必须大于等于字符串的实际长度+1

字符数组的定义和初始化

方式一：逐个字符初始化

char str[6] = {'H', 'e', 'l', 'l', 'o', '\0'};

方式二：直接用字符串常量初始化

char str[] = "Hello";

方式三：指定大小初始化

char str[20] = "Hello";

02 字符数组的输入输出

使用cin输入字符串：

char str[20];
cin >> str; // 不接收空格

使用cin.getline()输入含空格的字符串：

char str[100];
cin.getline(str, 100); // 可以接收空格

使用cout输出字符串：

char str[] = "Hello";
cout << str;

字符串输入输出示例

#include <iostream>
using namespace std;

int main() {
    char name[50];
    char address[100];
    
    cout << "请输入姓名：";
    cin >> name;
    
    cout << "请输入地址：";
    cin.ignore(); // 忽略换行符
    cin.getline(address, 100);
    
    cout << "姓名：" << name << endl;
    cout << "地址：" << address << endl;
    
    return 0;
}

03 字符串操作函数

C++提供了丰富的字符串操作函数，需要包含头文件<cstring>。


常用函数：

- strlen(str)：计算字符串长度（不含'\0'）

- strcpy(str1, str2)：将str2复制到str1

- strcat(str1, str2)：将str2追加到str1末尾

- strcmp(str1, str2)：比较str1和str2

- strstr(str1, str2)：在str1中查找str2

字符串操作函数示例

#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char str1[20] = "Hello";
    char str2[20] = " World";
    
    cout << "str1长度：" << strlen(str1) << endl;
    
    strcat(str1, str2);
    cout << "拼接后：" << str1 << endl;
    
    char str3[20];
    strcpy(str3, str1);
    cout << "复制后：" << str3 << endl;
    
    int cmp = strcmp(str1, str3);
    cout << "比较结果：" << cmp << endl;
    
    return 0;
}

04 字符串的遍历

使用for循环遍历：

char str[] = "Hello";
for (int i = 0; str[i] != '\0'; i++) {
    cout << str[i];
}

使用while循环遍历：

char str[] = "Hello";
int i = 0;
while (str[i] != '\0') {
    cout << str[i];
    i++;
}

字符串遍历示例

#include <iostream>
using namespace std;

int main() {
    char str[] = "Hello, C++!";
    
    int count = 0;
    for (int i = 0; str[i] != '\0'; i++) {
        count++;
    }
    cout << "字符个数：" << count << endl;
    
    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] >= 'a' && str[i] <= 'z') {
            str[i] -= 32;
        }
    }
    cout << "大写：" << str << endl;
    
    return 0;
}

编程题：密码合规检测

题目描述：

检查用户输入的密码是否合规。合规要求：

1. 只能由字母、数字、特殊字符组成

2. 长度在6~12之间

3. 至少包含一个特殊字符


输入格式：

输入一个字符串


输出格式：

合规输出"YES"，否则输出"NO"

密码合规检测参考代码

#include <iostream>
#include <cstring>
using namespace std;

bool check(char str[], int len) {
    if (len < 6 || len > 12) return false;
    bool hasSpecial = false;
    for (int i = 0; i < len; i++) {
        char c = str[i];
        if (!((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') ||
                (c >= '0' && c <= '9') || c == '!' || c == '@' ||
                c == '#' || c == '$' || c == '%' || c == '^')) {
            return false;
        }
        if (c == '!' || c == '@' || c == '#' || c == '$' ||
                c == '%' || c == '^') {
            hasSpecial = true;
        }
    }
    return hasSpecial;
}

int main() {
    char pwd[100];
    cin >> pwd;
    if (check(pwd, strlen(pwd))) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
    return 0;
}
